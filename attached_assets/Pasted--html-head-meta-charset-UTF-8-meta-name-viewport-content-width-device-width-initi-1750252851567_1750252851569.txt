<script>
document.getElementById('download-id-card-btn').addEventListener('click', function() {
    document.getElementById('download-options-modal').classList.remove('hidden');
});

document.getElementById('cancel-download').addEventListener('click', function() {
    document.getElementById('download-options-modal').classList.add('hidden');
});

document.getElementById('download-png').addEventListener('click', function() {
    document.getElementById('download-options-modal').classList.add('hidden');
    downloadAsImage();
});

document.getElementById('download-pdf').addEventListener('click', function() {
    document.getElementById('download-options-modal').classList.add('hidden');
    downloadAsPDF();
});

document.getElementById('download-qr-only').addEventListener('click', function() {
    document.getElementById('download-options-modal').classList.add('hidden');
    downloadQROnly();
});

function showProgress(text, progress) {
    document.getElementById('download-progress-modal').classList.remove('hidden');
    document.getElementById('progress-text').textContent = text;
    document.getElementById('progress-bar').style.width = progress + '%';
}

function hideProgress() {
    document.getElementById('download-progress-modal').classList.add('hidden');
}

function downloadAsImage() {
    showProgress('Capturing ID card...', 20);
    
    setTimeout(() => {
        showProgress('Processing image...', 60);
        
        setTimeout(() => {
            showProgress('Preparing download...', 90);
            
            setTimeout(() => {
                const element = document.getElementById('employee-id-card');
                html2canvas(element, {
                    backgroundColor: null,
                    scale: 2,
                    useCORS: true
                }).then(canvas => {
                    const link = document.createElement('a');
                    link.download = 'employee-id-card.png';
                    link.href = canvas.toDataURL();
                    link.click();
                    hideProgress();
                });
            }, 500);
        }, 800);
    }, 500);
}

function downloadAsPDF() {
    showProgress('Initializing PDF generation...', 10);
    
    setTimeout(() => {
        showProgress('Capturing ID card...', 30);
        
        setTimeout(() => {
            showProgress('Converting to PDF...', 70);
            
            setTimeout(() => {
                showProgress('Finalizing document...', 90);
                
                setTimeout(() => {
                    const element = document.getElementById('employee-id-card');
                    html2canvas(element, {
                        backgroundColor: null,
                        scale: 2,
                        useCORS: true
                    }).then(canvas => {
                        const { jsPDF } = window.jspdf;
                        const pdf = new jsPDF({
                            orientation: 'portrait',
                            unit: 'mm',
                            format: 'a4'
                        });
                        
                        const imgData = canvas.toDataURL('image/png');
                        const imgWidth = 180;
                        const imgHeight = (canvas.height * imgWidth) / canvas.width;
                        
                        pdf.addImage(imgData, 'PNG', 15, 20, imgWidth, imgHeight);
                        
                        pdf.setFontSize(16);
                        pdf.text('Employee ID Card - Attendance Tracker Pro', 15, 15);
                        
                        pdf.setFontSize(10);
                        pdf.text('Generated on: ' + new Date().toLocaleDateString(), 15, imgHeight + 30);
                        pdf.text('Employee: Sarah Johnson', 15, imgHeight + 40);
                        pdf.text('ID: #SJ002', 15, imgHeight + 50);
                        pdf.text('Position: Marketing Manager', 15, imgHeight + 60);
                        pdf.text('QR ID: #SJ002QR2024', 15, imgHeight + 70);
                        
                        pdf.save('employee-id-card.pdf');
                        hideProgress();
                    });
                }, 500);
            }, 800);
        }, 700);
    }, 500);
}

function downloadQROnly() {
    showProgress('Generating QR code...', 50);
    
    setTimeout(() => {
        showProgress('Preparing download...', 90);
        
        setTimeout(() => {
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            canvas.width = 400;
            canvas.height = 400;
            
            ctx.fillStyle = '#001F3F';
            ctx.fillRect(0, 0, 400, 400);
            
            ctx.fillStyle = 'white';
            ctx.font = '20px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('QR CODE', 200, 200);
            ctx.fillText('#SJ002QR2024', 200, 230);
            
            const link = document.createElement('a');
            link.download = 'qr-code.png';
            link.href = canvas.toDataURL();
            link.click();
            hideProgress();
        }, 500);
    }, 800);
}

    alert('New QR code generated successfully!');
});
