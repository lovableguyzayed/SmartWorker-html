<script>
document.addEventListener('DOMContentLoaded', function() {
    const navButtons = document.querySelectorAll('.nav-btn');
    const scannerBtn = document.getElementById('scanner-btn');
    const markAttendanceBtn = document.getElementById('mark-attendance-btn');
    const manageWorkersBtn = document.getElementById('manage-workers-btn');
    
    function updateTime() {
        const now = new Date();
        const timeString = now.toLocaleTimeString('en-US', {
            hour12: true,
            hour: '2-digit',
            minute: '2-digit',
            second: '2-digit'
        });
        document.getElementById('current-time').textContent = timeString;
    }
    
    function updateTotalPay() {
        const presentWorkers = 27;
        const dailyRate = 450;
        const overtimeHours = 8;
        const overtimeRate = 75;
        const totalPay = (presentWorkers * dailyRate) + (overtimeHours * overtimeRate);
        document.getElementById('total-pay').textContent = `₹${totalPay.toLocaleString()}`;
    }
    
    updateTime();
    updateTotalPay();
    setInterval(updateTime, 1000);
    setInterval(updateTotalPay, 30000);
    
    markAttendanceBtn.addEventListener('click', function() {
        this.classList.add('animate-pulse');
        setTimeout(() => {
            this.classList.remove('animate-pulse');
        }, 300);
        console.log('Opening Manual Attendance Marking...');
    });
    
    manageWorkersBtn.addEventListener('click', function() {
        this.classList.add('animate-pulse');
        setTimeout(() => {
            this.classList.remove('animate-pulse');
        }, 300);
        console.log('Opening Worker Management...');
    });
    
    navButtons.forEach(button => {
        button.addEventListener('click', function() {
            const tab = this.getAttribute('data-tab');
            
            navButtons.forEach(btn => {
                if (btn.id === 'scanner-btn') return;
                btn.classList.remove('text-accent');
                btn.classList.add('text-gray-400');
                const iconContainer = btn.querySelector('div');
                const span = btn.querySelector('span');
                if (iconContainer && !iconContainer.classList.contains('bg-blue-50')) {
                    iconContainer.classList.remove('bg-blue-50');
                }
                if (span) {
                    span.classList.remove('font-semibold');
                }
            });
            
            if (tab !== 'scanner') {
                this.classList.remove('text-gray-400');
                this.classList.add('text-accent');
                const iconContainer = this.querySelector('div');
                const span = this.querySelector('span');
                if (iconContainer) {
                    iconContainer.classList.add('bg-blue-50');
                }
                if (span) {
                    span.classList.add('font-semibold');
                }
            }
            
            if (tab === 'scanner') {
                scannerBtn.classList.add('animate-pulse');
                setTimeout(() => {
                    scannerBtn.classList.remove('animate-pulse');
                }, 300);
                console.log('Opening QR Scanner...');
            }
        });
    });
});
</script>
