<script>
document.addEventListener('DOMContentLoaded', function() {
    const signinForm = document.getElementById('signin-form');
    const signinBtn = document.getElementById('signin-btn');
    const otpBtn = document.getElementById('otp-btn');
    const passwordInput = document.querySelector('input[type="password"]');
    const eyeIcon = document.querySelector('.fa-eye');

    signinForm.addEventListener('submit', function(e) {
        e.preventDefault();
        signinBtn.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i>Signing In...';
        signinBtn.disabled = true;
        
        setTimeout(() => {
            console.log('Sign in successful');
        }, 2000);
    });

    otpBtn.addEventListener('click', function() {
        this.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i>Sending OTP...';
        this.disabled = true;
        
        setTimeout(() => {
            this.innerHTML = '<i class="fa-solid fa-mobile-alt text-accent mr-2"></i>Sign in with OTP';
            this.disabled = false;
        }, 2000);
    });

    if (eyeIcon) {
        eyeIcon.addEventListener('click', function() {
            if (passwordInput.type === 'password') {
                passwordInput.type = 'text';
                this.classList.remove('fa-eye');
                this.classList.add('fa-eye-slash');
            } else {
                passwordInput.type = 'password';
                this.classList.remove('fa-eye-slash');
                this.classList.add('fa-eye');
            }
        });
    }
});
</script>
