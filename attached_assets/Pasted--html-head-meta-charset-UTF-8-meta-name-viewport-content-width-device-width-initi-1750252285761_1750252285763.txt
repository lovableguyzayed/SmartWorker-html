<script>
document.addEventListener('DOMContentLoaded', function() {
    const addClosureBtn = document.getElementById('add-closure-btn');
    const closureDate = document.getElementById('closure-date');
    const closureRemark = document.getElementById('closure-remark');
    const allowAttendance = document.getElementById('allow-attendance');
    
    addClosureBtn.addEventListener('click', function() {
        if (closureDate.value && closureRemark.value) {
            // Add new closure date logic here
            closureDate.value = '';
            closureRemark.value = '';
            allowAttendance.checked = true;
        }
    });
    
    // Toggle functionality for switches
    const toggles = document.querySelectorAll('input[type="checkbox"]');
    toggles.forEach(toggle => {
        toggle.addEventListener('change', function() {
            const dot = this.parentElement.querySelector('.dot');
            const bg = this.parentElement.querySelector('div');
            
            if (this.checked) {
                dot.classList.add('translate-x-5');
                bg.classList.remove('bg-gray-300');
                bg.classList.add('bg-accent', 'bg-orange-500');
            } else {
                dot.classList.remove('translate-x-5');
                bg.classList.add('bg-gray-300');
                bg.classList.remove('bg-accent', 'bg-orange-500');
            }
        });
    });
});
</script>
