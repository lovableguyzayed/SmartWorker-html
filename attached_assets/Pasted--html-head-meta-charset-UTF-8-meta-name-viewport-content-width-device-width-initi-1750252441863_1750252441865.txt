<script>
document.addEventListener('DOMContentLoaded', function() {
    const tabBtns = document.querySelectorAll('.tab-btn');
    const filterBtns = document.querySelectorAll('.filter-btn');
    const workerCards = document.querySelectorAll('.worker-card');
    const searchInput = document.getElementById('search-workers');

    tabBtns.forEach(btn => {
        btn.addEventListener('click', function() {
            tabBtns.forEach(b => {
                b.classList.remove('active', 'bg-accent', 'text-white');
                b.classList.add('text-gray-600');
            });
            this.classList.add('active', 'bg-accent', 'text-white');
            this.classList.remove('text-gray-600');
        });
    });

    filterBtns.forEach(btn => {
        btn.addEventListener('click', function() {
            filterBtns.forEach(b => {
                b.classList.remove('bg-accent', 'text-white');
                b.classList.add('bg-white', 'text-gray-600', 'border', 'border-gray-100');
            });
            this.classList.add('bg-accent', 'text-white');
            this.classList.remove('bg-white', 'text-gray-600', 'border', 'border-gray-100');
            
            const filter = this.dataset.filter;
            filterWorkers(filter);
        });
    });

    function filterWorkers(filter) {
        workerCards.forEach(card => {
            if (filter === 'all') {
                card.style.display = 'block';
            } else if (filter === 'present' || filter === 'absent') {
                card.style.display = card.dataset.status === filter ? 'block' : 'none';
            } else if (filter === 'daily' || filter === 'fixed') {
                card.style.display = card.dataset.type === filter ? 'block' : 'none';
            }
        });
    }

    searchInput.addEventListener('input', function() {
        const searchTerm = this.value.toLowerCase();
        workerCards.forEach(card => {
            const workerName = card.querySelector('h3').textContent.toLowerCase();
            const position = card.querySelector('.text-gray-500').textContent.toLowerCase();
            if (workerName.includes(searchTerm) || position.includes(searchTerm)) {
                card.style.display = 'block';
            } else {
                card.style.display = 'none';
            }
        });
    });
});
</script>
