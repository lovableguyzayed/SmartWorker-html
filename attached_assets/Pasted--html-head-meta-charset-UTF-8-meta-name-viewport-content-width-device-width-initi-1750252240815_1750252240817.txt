<script>
document.addEventListener('DOMContentLoaded', function() {
    const signupForm = document.getElementById('signup-form');
    const signupBtn = document.getElementById('signup-btn');
    const otpSignupBtn = document.getElementById('otp-signup-btn');
    const passwordInputs = document.querySelectorAll('input[type="password"]');
    const eyeIcons = document.querySelectorAll('.fa-eye');

    signupForm.addEventListener('submit', function(e) {
        e.preventDefault();
        signupBtn.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i>Creating Account...';
        signupBtn.disabled = true;
        
        setTimeout(() => {
            console.log('Account created successfully');
        }, 2000);
    });

    otpSignupBtn.addEventListener('click', function() {
        this.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i>Sending OTP...';
        this.disabled = true;
        
        setTimeout(() => {
            this.innerHTML = '<i class="fa-solid fa-mobile-alt text-accent mr-2"></i>Sign up with OTP';
            this.disabled = false;
        }, 2000);
    });

    eyeIcons.forEach((eyeIcon, index) => {
        eyeIcon.addEventListener('click', function() {
            const passwordInput = passwordInputs[index];
            if (passwordInput.type === 'password') {
                passwordInput.type = 'text';
                this.classList.remove('fa-eye');
                this.classList.add('fa-eye-slash');
            } else {
                passwordInput.type = 'password';
                this.classList.remove('fa-eye-slash');
                this.classList.add('fa-eye');
            }
        });
    });
});
</script>
